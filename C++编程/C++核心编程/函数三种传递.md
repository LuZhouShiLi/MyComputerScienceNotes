# 函数三种传递

## 值传递

 ```cpp
// 值传递
void swap(int a, int b) {
	int tmp = a;
	a = b;
	b = tmp;
}

void print(int& a, int& b) {
	cout << a << " " << b << " " << endl;
}
int main() {
	// 普通引用
	
	int a = 10;
	int b = 20;
	swap(a, b);// 没有交换
	print(a, b);

	system("pause");
}
 ```


 ## 指针传递

```CPP
void swap1(int* a, int* b) {
	int tmp = *a;
	*a = *b;
	*b = tmp;
}
void print(int& a, int& b) {
	cout << a << " " << b << " " << endl;
}
int main() {
	// 普通引用
	
	int a = 10;
	int b = 20;
	swap1(&a, &b);// 发生交换
	print(a, b);

	system("pause");
}

```


## 引用传递

```cpp
// 引用传递
void swap2(int &a, int &b) {
	int tmp = a;
	a = b;
	b = tmp;
}


void print(int& a, int& b) {
	cout << a << " " << b << " " << endl;
}
int main() {
	// 普通引用
	
	int a = 10;
	int b = 20;
	swap2(a, b);// 交换
	print(a, b);

	system("pause");
}

```
